It may contain only stop words."} # Calculate cosine similarity between resume and each job role similarities = cosine_similarity(resume_vector, role_vectors).flatten() matched_jobs = [] for idx, score in enumerate(similarities): if score > 0.1: matched_jobs.append({ "role": job_roles[idx].get("role", "Unknown Role"), "accuracy": round(score * 100, 2) }) # Sort jobs by similarity score (accuracy) matched_jobs = sorted(matched_jobs, key=lambda x: x["accuracy"], reverse=True) matched_roles = [job["role"] for job in matched_jobs] # Collect all matched skills matched_skills = set() for job in job_roles: if job.get("role") in matched_roles: for skill in job.get("skills", []): matched_skills.add(skill.lower()) #
